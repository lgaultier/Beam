\section{Description du Modèle}
%1%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle{Caractéristique du modèle}
  \begin{itemize}
    \item Modèle NEMO de type canal, couplé avec LOBSTER pour la biogéochimie 
    \item Domaine : $160 \times 500 \times 4$ km ($28 \times 84 \times 30$ points de grilles)
    \item Résolution : 6 km 
    \item Structures sous-méso/mésoéchelles générées par un jet barocline instable. 

  \end{itemize}

\end{frame}
